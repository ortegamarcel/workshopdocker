# Schritt für Schritt Anleitung zu Übung 2 - Dockerfile - 20min

1. docker build -t workshopdocker/studentservice -f ./services/Dockerfile --build-arg JAR_FILE=/studentservice/student-0.1.jar --build-arg EXPOSE=8090 .
2. docker build -t workshopdocker/roomservice -f ./services/Dockerfile --build-arg JAR_FILE=/roomservice/room-0.1.jar --build-arg EXPOSE=8080 .
3. docker build -t workshopdocker/lectureservice -f ./services/Dockerfile --build-arg JAR_FILE=/lectureservice/lecture-0.1.jar --build-arg EXPOSE=8070 .
4. docker run -d --link mysql-container -P --name studentservice workshopdocker/studentservice
5. docker run -d --link mysql-container -P --name roomservice workshopdocker/roomservice
6. docker run -d --link mysql-container -P --name lectureservice workshopdocker/lectureservice
